BREAK_HOURS   = 5,6
MIN_BREAK     = 1
SLOT_DURATION = 1

CREATE TABLE TEACHLOAD(
	RANK        INT         NOT NULL,
	DESCRIPTION VARCHAR(30)         ,
	MIN_LOAD    INT         NOT NULL,
	MAX_LOAD    INT         NOT NULL
);

CREATE TABLE DEPARTMENT(
	CODE VARCHAR(5)  NOT NULL,
	NAME VARCHAR(30) NOT NULL
);

CREATE TABLE BLOCK(
	CODE VARCHAR(5) NOT NULL,
	NAME VARCHAR(30)        ,
	LOC  VARCHAR(30)
);

CREATE TABLE COURSE(
	CODE     VARCHAR(5)  NOT NULL,
	NAME     VARCHAR(30)        ,
	DURATION INT         NOT NULL
);

CREATE TABLE GROUP(
	COURSE   VARCHAR(20)  NOT NULL,
	YEAR     INT          NOT NULL,
	BRANCH   VARCHAR(20)  NOT NULL,
	G_NO     INT          NOT NULL,
	STRENGTH INT          NOT NULL,
	CODE     VARCHAR(20)  NOT NULL
);

CREATE TABLE SUBJECT(
	CODE       VARCHAR(20)  NOT NULL,
	DEPARTMENT VARCHAR(20)  NOT NULL,
	TYPE       VARCHAR(1)   NOT NULL,
	SLOT       INT          NOT NULL,
	DURATION   INT          NOT NULL,
	NAME       VARCHAR(20)          ,
	PROJECTOR  TINYINT(1)           ,
	ROOM
);

CREATE TABLE TEACHER(
	CODE       VARCHAR(4)   NOT NULL,
	DEPARTMENT VARCHAR(20)  NOT NULL,
	RANK       VARCHAR(20)  NOT NULL,
	LOAD       INT          NOT NULL,
	T_PREF     VARCHAR(72)  NOT NULL,
	S_PREF     VARCHAR(198) NOT NULL,
	PROJECTOR  TINYINT(1)           ,
	NAME       VARCHAR(30)
);

CREATE TABLE ROOM(
	R_NO       INT                 ,
	BLOCK      VARCHAR(1)  NOT NULL,
	TYPE       VARCHAR(1)          ,
	CAPACITY   INT                 ,
	BW_BOARD   TINYINT(1)          ,
	PROJECTOR  TINYINT(1)          ,
	DEPARTMENT VARCHAR(20)
);
